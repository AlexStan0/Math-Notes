        \item as $R(t) \in C[0, 1] \cap C^1(0, 1)$ then
        
        \[
            \begin{aligned}
                R^{\prime}\left(\frac{1}{2}\right) &= \frac{R(1) - R(0)}{1 - 0}\\
                &= \frac{11 - 0}{1} \\
                &= 11
            \end{aligned}    
        \]

        $\therefore R^{\prime}\left(\frac{1}{2}\right) = 11$ 

        \item 

        \[
            \begin{aligned}
                \int_0^1 R(t) dt &= (\frac{1}{3})(3) + (\frac{1}{3})(3) + (\frac{1}{3})(5)\\
                &= 1 + 1 + \frac{5}{3}\\
                &= \frac{6}{3} + \frac{5}{3}\\
                &= \frac{11}{3}
            \end{aligned}    
        \]

        \item 

        \[
            \begin{aligned}
                \int_0^{\frac{1}{3}} R^{\prime}(t) dt &= R\left(\frac{1}{3}\right) - R\left(0\right)\\
                &= 11 - 8\\
                &= 3
            \end{aligned}    
        \]
        
        \item  

        \[
            \begin{aligned}
                \sum_{k = 1}^n R\left(\frac{1}{4} + \frac{k}{2n}\right) \cdot \frac{1}{2n} &= \int_{\frac{1}{4}}^{\frac{3}{4}} R(t) dt
            \end{aligned}
        \]

    \end{enumerate}
